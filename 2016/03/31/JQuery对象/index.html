JQuery对象

  • JQuery

第一映像

在日常编程中可以感觉到，jQuery使程序变得精简。显然，jQuery对象与DOM对象是有区别的。因为jQuery对象能够操作DOM对象，我们知道jQuery对象中肯定包含DOM对象。\$(“p”).get(0).innerHTML可以获得第一个p元素的innerHTML，\$(“p”).get(0)取到的是一个DOM元素，不再是jQuery对象。jQuery的函数操作大多返回一个jQuery对象，当判断一个元素前面有没有别的元素时，使用pre()是否为undefined是无法判断的，因为就算没有元素，也会返回一个空的jQuery对象[object Object]，因此需要用pre().get(0)是否为undefined来判断前面是否还有元素。讲到这里有了个明确的概念：jQuery对象包含了DOM对象，并在jQuery对象中封装了各种对DOM对象进行操作的函数。

代码演变

一个简单的dom操作

首先，从一个很简单的功能开始说。一个按钮inp，点击一次，隐藏块div1。结构代码如下：

我们可能会这样写代码：

var div=document.getElementById("div1");
var btn=document.getElementById("div1");
btn.onclick=functioin(){
    div.style.display="none";
}

想偷懒的程序员们希望能够少写点getElementById这样长的句子，因此决定把得到dom元素的方法封装起来以便调用。

var F=function(id){
    return document.getElementById(id);
};
var i=F("inp");
var d=F("div1");
i.onclick=function(){
    d.style.display="none";
};

面向对象的设计

如果再想在F后链式实现.click()和.hide()，则需要使用构造函数。

var F = function(id) {
    return this.getElementById(id);
};
F.prototype.getElementById = function(id) {
    this.ele = document.getElementById(id);
    return this;
};
var i=new F("inp");
var d=new F("div1");
i.ele.onclick=function(){
    d.ele.style.display="none";
};

实现封装。将new F封装到\$中 将onclick和hide写为F的原型函数。

var F = function(id) {
    return this.getElementById(id);
};
F.prototype.getElementById = function(id) {
    this.ele = document.getElementById(id);
    return this;
};
F.prototype.click = function(fn) {
    this.ele.onclick=fn;
};
F.prototype.hide = function() {
    this.ele.style.display="none";
};
var $=function(id){
    return new F(id);
}
var i=$("inp");
var d=$("div1");
i.click(function(){
    d.hide();
});

这一步已经实现了\$(“”).click()和.hide()的功能。然而内部还是和jQuery原生颇有不同。

向jQuery格式靠近

jQuery中添加函数是使用\$.fn。因此，可以将F.prototype换成\$.fn试试：

var $=function(id){
    return new $.fn(id);
}
$.fn = function(id) {
    return this.getElementById(id);
};
$.fn.prototype.getElementById = function(id) {
    this.ele = document.getElementById(id);
    return this;
};
$.fn.prototype.click = function(fn) {
    this.ele.onclick=fn;
};
$.fn.prototype.hide = function() {
    this.ele.style.display="none";
};

由于并没有\$.fn这样的变量，因此\$.fn是\$上的一个属性，将\$的定义提前。
考虑到\$可以取得一堆node，且get(i)为第i个DOM元素。因此，直接将ele赋值到this[i]中。

var $=function(id){
    return new $.fn(id);
}
$.fn = function(id) {
    return this.init(id);
};
$.fn.prototype.init = function(selector,context) {//querySelectorAll ie8+
    var nodes=(context || document).querySelectorAll(selector);
    this.length=nodes.length;
    console.log(nodes.length);
    for(var i=0;i<this.length;++i){
        this[i]=nodes[i];
    }
    return this;//返回当前作用域，init只是在当前作用域上添加了nodes（真正可操作的dom元素）
};
$.fn.prototype.click = function(fn) {
    for(var i=0;i<this.length;++i){
        this[i].onclick=fn;
    }
};
$.fn.prototype.hide = function() {
    for(var i=0;i<this.length;++i){
        this[i].style.display="none";
    }
};

jQuery中添加函数应该是直接\$.fn.newFun这种形式，prototype应该被隐藏。因此，直接将\$.fn.prototype用$.fn1替代。new \$.fn(id)则可替换为new \$.fn1.init(id)。由于改变构造函数会改变对象的prototype导致返回的对象找不到click和hide函数，因此需要将\$.fn1.init.prototype设为\$.fn.prototype，也即是\$.fn1。

var $=function(id){
    return new $.fn1.init(id);
}
$.fn = function(id) {
    // return this.init(id);
};
$.fn1=$.fn.prototype;
$.fn1.init = function(selector,context) {
    var nodes=(context || document).querySelectorAll(selector);
    this.length=nodes.length;
    console.log(nodes.length);
    for(var i=0;i<this.length;++i){
        this[i]=nodes[i];
    }
    return this;
};
// 这样才能new $.fn1.init的对象能够有hide click函数
$.fn1.init.prototype=$.fn1;
$.fn1.click = function(fn) {
    for(var i=0;i<this.length;++i){
        this[i].onclick=fn;
    }
};
$.fn1.hide = function() {
    for(var i=0;i<this.length;++i){
        this[i].style.display="none";
    }
};
var i=$("#inp");
var d=$("#div1");
i.click(function(){
    d.hide();
});

用\$.fn替换\$.fn.prototype也是一样的，不知道怎么觉得用对象替换该对象的prototype有点奇怪，但其实是一样的，只是为了区分留下来的\$.fn因此多替换了一步。可以看到\$.fn 函数已经没有作用的，是个空函数，为了体现jQuery的精简，\$.fn1=\$.fn.prototype 设为一个空函数的prototype与设为函数$的prototype没有区别。
最后将\$.fn替换\$.fn1，删除\$.fn，设置\$.fn1=\$.prototype ，大功告成。

var $=function(id){
    return new $.fn.init(id);
}

$.fn=$.prototype;
$.fn.init = function(selector,context) {//querySelectorAll ie8以上支持
    var nodes=(context || document).querySelectorAll(selector);
    this.length=nodes.length;
    console.log(nodes.length);
    for(var i=0;i<this.length;++i){
        this[i]=nodes[i];
    }
    return this;//返回当前作用域，init只是在当前作用域上添加了nodes（真正可操作的dom元素）
};
// 这样才能new $.fn1.init的对象能够有hide click函数
$.fn.init.prototype=$.fn;
$.fn.click = function(fn) {
    for(var i=0;i<this.length;++i){
        this[i].onclick=fn;
    }
};
$.fn.hide = function() {
    for(var i=0;i<this.length;++i){
        this[i].style.display="none";
    }
};
var i=$("#inp");
var d=$("#div1");
i.click(function(){
    d.hide();
});
